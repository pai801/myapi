# One API 模型元数据 - 独立 Curl 命令集合
# 使用方法：直接复制 curl 命令并替换 BASE_URL 和 TOKEN 后执行

BASE_URL="http://localhost:3000"
TOKEN="YOUR_ADMIN_TOKEN"

# ==================== 查看已导入的模型元数据 ====================

# 获取所有模型元数据
curl -X GET "${BASE_URL}/api/model-metadata/" -H "Authorization: Bearer ${TOKEN}"

# 获取单个模型元数据
curl -X GET "${BASE_URL}/api/model-metadata/gpt4turbo" -H "Authorization: Bearer ${TOKEN}"
